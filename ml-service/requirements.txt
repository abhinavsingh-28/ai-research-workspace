# ============================================
# AI Research Workspace — ML Service Dependencies
# ============================================
# Install all of these with: pip install -r requirements.txt
# Always run this AFTER activating the virtual environment.

# --- Web Framework ---
# FastAPI: the web framework that handles incoming HTTP requests.
# uvicorn: the server that actually runs FastAPI (FastAPI is just the framework,
#          uvicorn is the engine that listens for requests and passes them to FastAPI).
fastapi==0.115.0
uvicorn==0.30.6

# --- PDF Parsing ---
# PyMuPDF: reads PDF files and extracts the text from each page.
# The package name on PyPI is "PyMuPDF" but you import it as "fitz" in code.
PyMuPDF==1.24.10

# --- AI / RAG ---
# langchain: the framework that orchestrates the RAG pipeline
#            (chunking, embedding, retrieval, prompt building).
# langchain-google-genai: the LangChain plugin that connects to Google's Gemini API.
# langchain-chroma: the LangChain plugin that connects to ChromaDB.
langchain==0.3.1
langchain-google-genai==2.0.0
langchain-chroma==0.1.4

# --- Vector Store ---
# chromadb: the database that stores document embeddings (vectors)
#           and lets us search for similar vectors.
chromadb==0.5.5

# --- Utilities ---
# python-dotenv: reads the .env file so we can use environment variables
#               (like API keys) without hardcoding them.
# python-multipart: allows FastAPI to accept file uploads (multipart form data).
python-dotenv==1.0.1
python-multipart==0.0.9
